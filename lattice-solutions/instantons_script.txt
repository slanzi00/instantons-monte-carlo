reset
fname4 = "instanton_density_1_4.txt"
fname5 = "instanton_density_1_5.txt"
fname6 = "instanton_density_1_6.txt"

set autoscale xfixmin 
set autoscale xfixmax
set logscale xy
set mxtics 10
set ytics 5
set mytics 2

set style data linespoints
set multiplot layout 1,3 columns

# plotting number of instantons
set ylabel 'N_{inst}'
set xlabel 'n_{cool}'
plot fname4 u 1:2:3 index 0 w yerrorbars lc rgb 'blue' pt 4 t "{/Symbol h} = 1.4", \
     fname4 u 1:4 index 0 w l lc rgb 'orange' notitle, \
     fname4 u 1:5 index 0 w l lc rgb 'red' notitle, \
     fname5 u 1:2:3 index 0 w yerrorbars lc rgb 'blue' pt 6 t "{/Symbol h} = 1.5", \
     fname5 u 1:4 index 0 w l lc rgb 'orange' notitle, \
     fname5 u 1:5 index 0 w l lc rgb 'red' notitle, \
     fname6 u 1:2:3 index 0 w yerrorbars lc rgb 'blue' pt 8 t "{/Symbol h} = 1.6", \
     fname6 u 1:4 index 0 w l lc rgb 'orange' notitle, \
     fname6 u 1:5 index 0 w l lc rgb 'red' notitle

# plotting action
set ylabel 'S_{inst}'
set xlabel 'n_{cool}'
plot fname4 u 1:2:3 index 1 w yerrorbars lc rgb 'blue' pt 4 t "{/Symbol h} = 1.4", \
     fname4 u 1:4 index 1 w l lc rgb 'green' notitle, \
     fname5 u 1:2:3 index 1 w yerrorbars lc rgb 'blue' pt 6 t "{/Symbol h} = 1.5", \
     fname5 u 1:4 index 1 w l lc rgb 'green' notitle, \
     fname6 u 1:2:3 index 1 w yerrorbars lc rgb 'blue' pt 8 t "{/Symbol h} = 1.6", \
     fname6 u 1:4 index 1 w l lc rgb 'green' notitle

# plotting action per instanton
set ylabel 'S/N_{inst}'
set xlabel 'n_{cool}'
plot fname4 u 1:2:3 index 2 w yerrorbars lc rgb 'blue' pt 4 t "{/Symbol h} = 1.4", \
     fname4 u 1:4 index 2 w l lc rgb 'green' notitle, \
     fname5 u 1:2:3 index 2 w yerrorbars lc rgb 'blue' pt 6 t "{/Symbol h} = 1.5", \
     fname5 u 1:4 index 2 w l lc rgb 'green' notitle, \
     fname6 u 1:2:3 index 2 w yerrorbars lc rgb 'blue' pt 8 t "{/Symbol h} = 1.6", \
     fname6 u 1:4 index 2 w l lc rgb 'green' notitle

unset multiplot
